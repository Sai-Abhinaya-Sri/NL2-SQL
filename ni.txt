def get_all_tables_schema(connection):
    """
    ✅ FIXED: Determine DB type from connection class, not from session state.
    This allows calling the function for either MySQL or PostgreSQL correctly.
    """
    db_type = 'PostgreSQL' if isinstance(connection, psycopg2.extensions.connection) else 'MySQL'

    try:
        cursor = connection.cursor()
        excluded_tables = config.DATABASE_CONFIG.get('excluded_tables', [])
        max_sample_rows = config.DATABASE_CONFIG.get('max_sample_rows', 3)

        if db_type == 'PostgreSQL':
            cursor.execute(
                "SELECT table_name FROM information_schema.tables "
                "WHERE table_schema = 'public' AND table_type = 'BASE TABLE'"
            )
        else:
            cursor.execute("SHOW TABLES")

        tables = cursor.fetchall()

        all_schemas = {}

        for table_tuple in tables:
            table_name = table_tuple[0]

            if table_name in excluded_tables:
                continue

            if db_type == 'PostgreSQL':
                cursor.execute(
                    "SELECT column_name, data_type FROM information_schema.columns "
                    "WHERE table_name = %s ORDER BY ordinal_position",
                    (table_name,)
                )
                columns = cursor.fetchall()
                cursor.execute(f'SELECT * FROM "{table_name}" LIMIT %s', (max_sample_rows,))
                sample_data = cursor.fetchall()
                cursor.execute(f'SELECT COUNT(*) FROM "{table_name}"')
            else:
                cursor.execute(f"DESCRIBE `{table_name}`")
                columns = cursor.fetchall()
                cursor.execute(f"SELECT * FROM `{table_name}` LIMIT {max_sample_rows}")
                sample_data = cursor.fetchall()
                cursor.execute(f"SELECT COUNT(*) FROM `{table_name}`")

            row_count = cursor.fetchone()[0]

            all_schemas[table_name] = {
                'columns': [{'name': col[0], 'type': col[1]} for col in columns],
                'sample_data': sample_data,
                'row_count': row_count
            }

        cursor.close()
        return all_schemas, None

    except Exception as e:
        return {}, f"Error fetching schema: {str(e)}"


def nl_to_sql_groq(natural_language_query, tables_info):
    """
    ✅ FIXED: Determine DB type by detecting which schema bucket was used.
    If tables come from postgres_tables_info, use PostgreSQL syntax.
    If tables come from mysql_tables_info, use MySQL syntax.
    If mixed, detect from table names or default to MySQL.
    """
    api_key = config.GROQ_API_CONFIG['api_key']

    if api_key == 'YOUR_GROQ_API_KEY_HERE':
        raise Exception("Please set your Groq API key in config.py")

    # ✅ Determine DB type from selected tables
    # If PostgreSQL tables are selected, use PostgreSQL syntax
    postgres_selected = set(st.session_state.postgres_selected_tables)
    tables_are_pg = any(t in postgres_selected for t in tables_info.keys())
    is_pg = tables_are_pg

    schema_description = f"{'PostgreSQL' if is_pg else 'MySQL'} Database Schema - USE THESE EXACT COLUMN NAMES:\n\n"

    for table_name, info in tables_info.items():
        if is_pg:
            schema_description += f'TABLE: "{table_name}" ({info["row_count"]} rows)\n'
            schema_description += 'COLUMNS (use exactly as shown):\n'
            for col in info['columns']:
                schema_description += f'  - "{col["name"]}" ({col["type"]})\n'
            if info['sample_data'] and len(info['sample_data']) > 0:
                schema_description += '\nSample row:\n'
                sample_row = info['sample_data'][0]
                for idx, col in enumerate(info['columns']):
                    if idx < len(sample_row):
                        schema_description += f'  "{col["name"]}" = {sample_row[idx]}\n'
        else:
            schema_description += f"TABLE: `{table_name}` ({info['row_count']} rows)\n"
            schema_description += 'COLUMNS (use exactly as shown):\n'
            for col in info['columns']:
                schema_description += f"  - `{col['name']}` ({col['type']})\n"
            if info['sample_data'] and len(info['sample_data']) > 0:
                schema_description += '\nSample row:\n'
                sample_row = info['sample_data'][0]
                for idx, col in enumerate(info['columns']):
                    if idx < len(sample_row):
                        schema_description += f"  `{col['name']}` = {sample_row[idx]}\n"

        schema_description += '\n' + '='*50 + '\n\n'

    if is_pg:
        prompt = (
            'You are a PostgreSQL SQL expert. Generate SQL using ONLY the exact column names shown below.\n\n'
            + schema_description +
            f'\nUser Question: "{natural_language_query}"\n\n'
            'CRITICAL RULES:\n'
            '1. Use ONLY column names listed above\n'
            '2. NEVER use backticks (`) - PostgreSQL does NOT support them\n'
            '3. Use unquoted identifiers for normal table/column names\n'
            '4. Use single quotes for string values\n'
            '5. Return ONLY executable SQL - no markdown, no ```sql blocks\n'
            '6. Use proper JOINs when multiple tables needed\n'
            '7. For top N queries, use ORDER BY ... DESC LIMIT N\n'
            '8. ALWAYS close all string literals\n'
            'Generate the SQL query now:'
        )
        system_msg = ('You are a PostgreSQL SQL expert. NEVER use backticks. '
                      'Use unquoted or double-quoted identifiers. '
                      'Single quotes for string values. Return ONLY executable SQL.')
    else:
        prompt = (
            'You are a MySQL SQL expert. Generate SQL using ONLY the exact column names shown below.\n\n'
            + schema_description +
            f'\nUser Question: "{natural_language_query}"\n\n'
            'CRITICAL RULES:\n'
            '1. Use ONLY column names listed above\n'
            '2. ALWAYS use backticks (`) for ALL table and column names\n'
            '3. Return ONLY executable SQL - no markdown, no ```sql blocks\n'
            '4. Use proper JOINs when multiple tables needed\n'
            '5. For top N queries, use ORDER BY ... DESC LIMIT N\n'
            '6. ALWAYS close all string literals\n'
            'Generate the SQL query now:'
        )
        system_msg = ('You are a MySQL SQL expert. Always use backticks ` for identifiers. Return ONLY executable SQL.')

    url = 'https://api.groq.com/openai/v1/chat/completions'
    headers = {'Content-Type': 'application/json', 'Authorization': f'Bearer {api_key}'}
    data = {
        'messages': [
            {'role': 'system', 'content': system_msg},
            {'role': 'user', 'content': prompt}
        ],
        'model': config.GROQ_API_CONFIG['model_name'],
        'temperature': config.GROQ_API_CONFIG['temperature'],
        'max_tokens': config.GROQ_API_CONFIG['max_tokens']
    }

    try:
        response = requests.post(url, headers=headers, json=data, timeout=config.GROQ_API_CONFIG['timeout'])
        response.raise_for_status()
        result = response.json()
        sql_query = result['choices'][0]['message']['content'].strip()
        sql_query = sql_query.replace('```sql', '').replace('```', '').strip()
        sql_query = sql_query.strip('"').strip("'")

        if is_pg:
            sql_query = sql_query.replace('`', '')  # strip stray backticks

        usage = result.get('usage', {})
        token_usage = {
            'prompt_tokens': usage.get('prompt_tokens', 0),
            'completion_tokens': usage.get('completion_tokens', 0),
            'total_tokens': usage.get('total_tokens', 0),
        }

        return sql_query, token_usage

    except Exception as e:
        raise Exception(f"Groq API Error: {str(e)}")


def fix_sql_syntax(sql_query):
    """
    ✅ FIXED: Determine DB type from selected tables instead of session state.
    """
    postgres_selected = set(st.session_state.postgres_selected_tables)
    tables_are_pg = any(t in postgres_selected for t in [sql_query])  # Fallback
    is_pg = tables_are_pg if tables_are_pg else len(st.session_state.postgres_selected_tables) > 0

    if is_pg:
        sql_query = sql_query.replace('`', '')  # no backticks in PostgreSQL
        import re as _re
        sql_query = _re.sub(r'\[([^\]]+)\]', r'"\1"', sql_query)
    else:
        import re as _re
        sql_query = _re.sub(r'\[([^\]]+)\]', r'`\1`', sql_query)

    stripped = sql_query.rstrip().rstrip(';').rstrip()
    quote_count = stripped.count("'")

    if quote_count % 2 != 0:
        had_semicolon = sql_query.rstrip().endswith(';')
        sql_query = stripped + "'" + (";" if had_semicolon else "")

    return sql_query

def validate_single_db(tables_info):
    mysql_tables = set(st.session_state.mysql_tables_info.keys())
    postgres_tables = set(st.session_state.postgres_tables_info.keys())

    uses_mysql = any(t in mysql_tables for t in tables_info)
    uses_pg = any(t in postgres_tables for t in tables_info)

    if uses_mysql and uses_pg:
        return False
    return True

UNSAFE_PATTERNS = re.compile(
    r'\b(DROP|DELETE|TRUNCATE|ALTER|UPDATE|INSERT|CREATE|RENAME|REPLACE|GRANT|REVOKE)\b',
    re.IGNORECASE
)

def validate_sql_safety(sql_query: str):
    match = UNSAFE_PATTERNS.search(sql_query)
    if match:
        return False, match.group(0).upper()
    return True, None

def get_relevant_tables(nl_query: str, tables_info: dict, top_n: int = 5) -> dict:
    if not tables_info:
        return tables_info
    query_tokens = set(re.findall(r'\w+', nl_query.lower()))
    scores = {}
    for table_name, info in tables_info.items():
        score = 0
        for token in re.findall(r'\w+', table_name.lower()):
            if token in query_tokens:
                score += 2
        for col in info['columns']:
            for token in re.findall(r'\w+', col['name'].lower()):
                if token in query_tokens:
                    score += 1
        scores[table_name] = score
    relevant = {t: tables_info[t] for t, s in scores.items() if s > 0}
    if relevant:
        top_tables = sorted(relevant, key=lambda t: scores[t], reverse=True)[:top_n]
        return {t: tables_info[t] for t in top_tables}
    try:
        api_key = config.GROQ_API_CONFIG['api_key']
        table_names = list(tables_info.keys())
        fallback_prompt = (
            f"Given these table names: {table_names}\n"
            f"Which tables are most relevant to answer: \"{nl_query}\"?\n"
            f"Reply with ONLY a comma-separated list of table names, nothing else."
        )
        url = "https://api.groq.com/openai/v1/chat/completions"
        headers = {"Content-Type": "application/json", "Authorization": f"Bearer {api_key}"}
        data = {
            "messages": [{"role": "user", "content": fallback_prompt}],
            "model": config.GROQ_API_CONFIG['model_name'],
            "temperature": 0,
            "max_tokens": 100
        }
        resp = requests.post(url, headers=headers, json=data, timeout=config.GROQ_API_CONFIG['timeout'])
        resp.raise_for_status()
        raw = resp.json()['choices'][0]['message']['content'].strip()
        chosen = [t.strip().strip('`') for t in raw.split(',')]
        filtered = {t: tables_info[t] for t in chosen if t in tables_info}
        if filtered:
            return filtered
    except Exception:
        pass
    return tables_info

def get_user_selected_tables() -> dict:
    """
    ✅ CORRECT: Merge selected tables from BOTH databases.
    """
    merged_tables = {}
    
    # Add selected MySQL tables
    mysql_tables = st.session_state.mysql_selected_tables
    for table_name in mysql_tables:
        if table_name in st.session_state.mysql_tables_info:
            merged_tables[table_name] = st.session_state.mysql_tables_info[table_name]
    
    # Add selected PostgreSQL tables
    postgres_tables = st.session_state.postgres_selected_tables
    for table_name in postgres_tables:
        if table_name in st.session_state.postgres_tables_info:
            merged_tables[table_name] = st.session_state.postgres_tables_info[table_name]
    
    return merged_tables

def get_sqlalchemy_engine(relevant_tables: dict):
    postgres_tables = set(st.session_state.postgres_tables_info.keys())
    
    # If ALL tables belong to postgres → use postgres
    if all(t in postgres_tables for t in relevant_tables.keys()):
        cfg = config.POSTGRES_CONFIG
        return create_engine(
            f"postgresql+psycopg2://{cfg['user']}:{cfg['password']}@{cfg['host']}:{cfg['port']}/{cfg['database']}"
        )
    
    # Else → MySQL
    cfg = config.MYSQL_CONFIG
    return create_engine(
        f"mysql+mysqlconnector://{cfg['user']}:{cfg['password']}@{cfg['host']}:{cfg['port']}/{cfg['database']}"
    )

def execute_sql_query(connection, sql_query):
    engine = get_sqlalchemy_engine(relevant_tables)
    try:
        df = pd.read_sql_query(sql_query, engine)
        return df, None, sql_query
    except Exception as e:
        try:
            fixed_sql = fix_sql_syntax(sql_query)
            df = pd.read_sql_query(fixed_sql, engine)
            return df, "✅ Query auto-fixed (converted [] to ` for MySQL)", fixed_sql
        except Exception as e2:
            return None, str(e2), sql_query

def create_visualization(df, viz_type):
    if df is None or df.empty:
        return None
    numeric_cols = df.select_dtypes(include=['number']).columns.tolist()
    categorical_cols = df.select_dtypes(include=['object', 'category']).columns.tolist()
    try:
        if viz_type == "Bar Chart":
            if categorical_cols and numeric_cols:
                fig = px.bar(df, x=categorical_cols[0], y=numeric_cols[0],
                             title=f"{numeric_cols[0]} by {categorical_cols[0]}",
                             color=numeric_cols[0], color_continuous_scale='Blues')
                fig.update_layout(showlegend=False, height=500)
                return fig
        elif viz_type == "Line Chart":
            if len(df) > 0:
                x_col = df.columns[0]
                y_col = numeric_cols[0] if numeric_cols else df.columns[1]
                fig = px.line(df, x=x_col, y=y_col, title=f"{y_col} over {x_col}", markers=True)
                fig.update_traces(line_color='#1f77b4', line_width=3)
                fig.update_layout(height=500)
                return fig
        elif viz_type == "Pie Chart":
            if categorical_cols and numeric_cols:
                fig = px.pie(df, names=categorical_cols[0], values=numeric_cols[0],
                             title=f"Distribution of {numeric_cols[0]} by {categorical_cols[0]}", hole=0.3)
                fig.update_traces(textposition='inside', textinfo='percent+label')
                fig.update_layout(height=500)
                return fig
        elif viz_type == "Scatter Plot":
            if len(numeric_cols) >= 2:
                fig = px.scatter(df, x=numeric_cols[0], y=numeric_cols[1],
                                 title=f"{numeric_cols[1]} vs {numeric_cols[0]}",
                                 color=categorical_cols[0] if categorical_cols else None)
                fig.update_traces(marker=dict(size=10, opacity=0.7))
                fig.update_layout(height=500)
                return fig
        elif viz_type == "Histogram":
            if numeric_cols:
                fig = px.histogram(df, x=numeric_cols[0],
                                   title=f"Distribution of {numeric_cols[0]}",
                                   nbins=20, color_discrete_sequence=['#1f77b4'])
                fig.update_layout(height=500, showlegend=False)
                return fig
        elif viz_type == "Box Plot":
            if numeric_cols:
                if categorical_cols:
                    fig = px.box(df, x=categorical_cols[0], y=numeric_cols[0],
                                 title=f"{numeric_cols[0]} by {categorical_cols[0]}",
                                 color=categorical_cols[0])
                else:
                    fig = px.box(df, y=numeric_cols[0],
                                 title=f"Box Plot of {numeric_cols[0]}",
                                 color_discrete_sequence=['#1f77b4'])
                fig.update_layout(height=500)
                return fig
        elif viz_type == "Heatmap":
            if len(numeric_cols) >= 2:
                corr = df[numeric_cols].corr()
                fig = px.imshow(corr, labels=dict(color="Correlation"),
                                x=corr.columns, y=corr.columns,
                                title="Correlation Heatmap",
                                color_continuous_scale='RdBu_r', aspect='auto')
                fig.update_layout(height=500)
                return fig
    except Exception as e:
        print(f"Visualization error: {e}")
        return None
    return None

def suggest_best_chart(df):
    if df is None or df.empty:
        return "Table"
    numeric_cols = df.select_dtypes(include=['number']).columns.tolist()
    categorical_cols = df.select_dtypes(include=['object', 'category']).columns.tolist()
    if len(df) <= 1:
        return "Table"
    if categorical_cols and numeric_cols and len(df) <= 20:
        return "Bar Chart"
    if categorical_cols and numeric_cols:
        unique_cats = df[categorical_cols[0]].nunique()
        if unique_cats <= 10 and len(df) <= 50:
            return "Pie Chart"
    if len(numeric_cols) >= 2:
        return "Scatter Plot"
    if len(df) > 5:
        return "Line Chart"
    return "Table"

def is_zero_aggregate(df) -> bool:
    """Return True when df is a single-row, all-numeric result whose values are all 0."""
    if df is None or df.empty or len(df) != 1:
        return False
    numeric_cols = df.select_dtypes(include=['number']).columns.tolist()
    text_cols = df.select_dtypes(include=['object', 'category']).columns.tolist()
    if not numeric_cols or text_cols:
        return False
    return all(df[col].iloc[0] == 0 for col in numeric_cols)


def extract_search_terms(nl_query: str) -> list:
    """Extract potential search terms (names, identifiers) from the query"""
    quoted = re.findall(r'["\']([^"\']+)["\']', nl_query)
    if quoted:
        return quoted
    stop_words = {
        'show', 'find', 'get', 'list', 'select', 'display', 'give', 'tell',
        'what', 'where', 'when', 'how', 'all', 'the', 'me', 'my', 'a', 'an',
        'of', 'in', 'by', 'for', 'from', 'who', 'which', 'whose', 'top',
        'total', 'count', 'sum', 'average', 'number', 'many'
    }
    words = nl_query.split()
    capitalized = [
        w.strip('.,!?') for w in words
        if w and w[0].isupper() and w.strip('.,!?').lower() not in stop_words
    ]
    return capitalized


def pre_check_entity_exists(search_terms: list, tables_info: dict, connection) -> tuple:
    """Quick scan: check if any search term exists in any text column."""
    if not search_terms or connection is None:
        return True, None

    try:
        cursor = connection.cursor()
        for term in search_terms:
            term_found = False
            for table_name, info in tables_info.items():
                text_cols = [
                    col['name'] for col in info['columns']
                    if any(t in col['type'].lower() for t in ('char', 'text', 'enum', 'varchar'))
                ]
                for col in text_cols:
                    try:
                        cursor.execute(
                            f"SELECT 1 FROM `{table_name}` WHERE `{col}` LIKE %s LIMIT 1",
                            (f"%{term}%",)
                        )
                        if cursor.fetchone():
                            term_found = True
                            break
                    except Exception:
                        continue
                if term_found:
                    break
            if not term_found:
                cursor.close()
                return False, term
        cursor.close()
    except Exception:
        return True, None
    return True, None
